#Function for calculating Distance
def calculateDistance(lat1, lon1, lat2, lon2):
    R = 6373.0
    dlon =radians(lon2 - lon1)
    dlat =radians(lat2 - lat1)
    lat1 =radians(lat1)
    lat2 =radians(lat2)
    a = (sin(dlat/2))**2 + cos(lat1) * cos(lat2) * (sin(dlon/2))**2
    c = 2 * atan2(sqrt(a), sqrt(1-a))
    distance = R * c
    return distance #Returns Distance in KM

#Function for calculating Gradient
def calculateGradient(elev1,elev2):
    gradient=elev2-elev1
    return gradient

#Function for calculating Time
def calculateTime(Time1,Time2):
    Time=Time2-Time1
    Time=Time.total_seconds() #Convert the deltatime to seconds
    return Time/3600          #Convert to Hours,return time in Seconds
 
#Function for Calculating Speed    
def calculateSpeed(Distance1,Time1):
    Speed =Distance1/Time1
    return Speed  #returns Speed in KM/HR
